189 CONSTEXPR14 auto to_decimal_using_switch(const char* csd) -> double {
190 // Handle integral part using int (bit shifts are faster than FP ops)
191 auto integral = 0;
192 for (; *csd != '.' && *csd != '\0'; ++csd) {
193 switch (*csd) {
194 case '0':
195 integral *= 2;
196 break;
197 case '+':
198 integral = integral * 2 + 1;
199 break;
200 case '-':
201 integral = integral * 2 - 1;
202 break;
203 default:
204 throw std::invalid_argument("Work with 0, +, -, and . only");
205 }
206 }
207 if (*csd == '\0') {
208 return double(integral);
209 }
210 // Handle fractional part
211 auto decimal_value = double(integral);
212 auto scale = 0.5;
213 for (++csd; *csd != '\0'; ++csd) {
214 switch (*csd) {
215 case '0':
216 break;
217 case '+':
218 decimal_value += scale;
219 break;
220 case '-':
221 decimal_value -= scale;
222 break;
223 default:
224 throw std::invalid_argument("Fractional part work with 0, +, and - only");
225 }
226 scale /= 2;
227 }
228 return decimal_value;
229 }
